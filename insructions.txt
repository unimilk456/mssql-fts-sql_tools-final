1) clone project

2)
cd .. mssql-fts-sql_tools-final

3) build image
docker build --platform=linux/amd64 --no-cache -t mssql-full . 2>&1 | tee build2.log

4) wait about 40 seconds and run Spring app
#Replace super@123 to your mssql container password
docker run -e ACCEPT_EULA=Y -e SA_PASSWORD=super@123 -e DB_NAME=eboard-local -p 1433:1433 --restart=no -d mssql-full:latest
